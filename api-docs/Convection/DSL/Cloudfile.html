def stack(stack_name, template, options = {}, &block)
  options[:region] ||= region
  options[:cloud] = name
  options[:attributes] = attributes

  @stacks[stack_name] = Control::Stack.new(stack_name, template, options, &block)
  @deck << @stacks[stack_name]
end
