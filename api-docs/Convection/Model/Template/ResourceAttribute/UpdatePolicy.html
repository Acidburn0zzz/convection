def render(resource)
  resource.tap do |r|
    r['UpdatePolicy'] = {
      'AutoScalingScheduledAction' => {
        'IgnoreUnmodifiedGroupSizeProperties' => true
      },
      'AutoScalingRollingUpdate' => {
        'MinInstancesInService' => @min_in_service,
        'MaxBatchSize' => @max_batch,
        'WaitOnResourceSignals' => false,
        'PauseTime' => @pause
      }
    }
  end
end
