def map(no_nil = false, &block)
  result = {}

  each do |key, value|
    res = block.call(value)

    next if no_nil && res.nil?
    next if no_nil && res.is_a?(Array) && res.empty?
    next if no_nil && res.is_a?(Hash) && res.empty?

    result[key] = res
  end

  result
end
